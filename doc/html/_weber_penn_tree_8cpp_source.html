585  vec3 pvec = cross( parent_normal, lnorm );
586 
587  //rotate leaf based on downangle
588  if( downangle<0.f ){
589  downangle = M_PI+downangle;
590  }
591  context->getObjectPointer(ID)->rotate( -downangle, pvec );
592 
593  //shift leaf perpendicular to the direction of the branch
594  vec3 offset = -0.6*parameters.LeafScale*scale*fabs(sinf(downangle))*lnorm;
595 
596  context->getObjectPointer(ID)->translate( offset );
597 
598  //random azimuthal rotation about the branch
599  context->getObjectPointer(ID)->rotate( phi, parent_normal );
600 
601  //translate to the position of the branch
602  context->getObjectPointer(ID)->translate(position);
603 
604  }
605 
606  }

610 float WeberPennTree::ShapeRatio( uint shape, float ratio ){
611 
612  if( shape==0 ){ //conical
613  return 0.2+0.8*ratio;
614  }else if( shape==1 ){ //spherical
615  return 0.2+0.8*sin(M_PI*ratio);
616  }else if( shape==2 ){ //hemispherical
617  return 0.2+0.8*sin(0.5*M_PI*ratio);
618  }else if( shape==3 ){ //cylindrical
619  return 1.f;
620  }else if( shape==4 ){ //tapered cylinder
621  return 0.5+0.5*ratio;
622  }else if( shape==5 ){ //flame
623  if( ratio<=0.7 ){
624  return ratio/0.7f;
625  }else{
626  return (1.f-ratio)/0.3f;
627  }
628  }else if( shape==6 ){ //inverse conical
629  return 1-0.8*ratio;
630  }else if( shape==7 ){ //tend flame
631  if( ratio<=0.7 ){
632  return 0.5+0.5*ratio/0.7f;
633  }else{
634  return 0.5+0.5*(1-ratio)/0.3f;
635  }
636  }else if( shape==8 ){ //envelope
637  return 0;
638  }
639 
640  return 0;
641
